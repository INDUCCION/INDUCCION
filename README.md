    <script>
        const videos = [
            { src: 'INDUCCION IDENTIDAD CORPORATIVA.mp4' },
            { src: 'sst.mp4' },
            { src: 'asovinos.mp4' },
            { src: 'gestion ambiental.mp4' },
            { src: 'fedi.mp4' },
            { src: 'control interno.mp4' },
            { src: 'sagrilaft.mp4' },
            { src: 'dan.mp4' }
        ];

        let currentVideo = 0;
        const videoPlayer = document.getElementById('videoPlayer');
        const videoSource = document.getElementById('videoSource');
        const nextButton = document.getElementById('nextButton');
        const qrContainer = document.getElementById('qrContainer');
        const welcomeContainer = document.getElementById('welcomeContainer');
        const videoContainer = document.getElementById('videoContainer');

        // Función para iniciar la inducción (oculta la página de bienvenida y muestra los videos)
        function startInduction() {
            welcomeContainer.style.display = 'none';
            videoContainer.style.display = 'block';
            loadVideo(0); // Cargar el primer video
        }

        // Función para cargar un video
        function loadVideo(index) {
            videoSource.src = videos[index].src;
            videoPlayer.load();
        }

        // Cuando termina el video, mostrar el botón "Siguiente"
        videoPlayer.onended = function() {
            if (currentVideo < videos.length - 1) {
                nextButton.style.display = 'block';
            } else {
                window.location.href = '#qrContainer'; // Llevar a la página final
                qrContainer.style.display = 'flex';
            }
        };

        // Función para avanzar al siguiente video
        function nextVideo() {
            currentVideo++;
            if (currentVideo < videos.length) {
                loadVideo(currentVideo);
                nextButton.style.display = 'none'; // Ocultar el botón hasta que termine el siguiente video
            } else {
                // Mostrar la página con el QR y enlace
                qrContainer.style.display = 'flex';
            }
        }

        // Cargar el primer video al iniciar
        loadVideo(0);
    </script>
